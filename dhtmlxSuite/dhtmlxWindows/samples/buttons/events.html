Sample: dhtmlxWindows Buttons Events dhtmlxWindows main page
Catch "onHelp" event
Redefine default "Park" event		 
 
<script>
 
    var dhxWins = new dhtmlXWindows();
    ...
    // attach "onHelp" event to window
    dhxWins.window(id).attachEvent("onHelp", function() {
        alert('button "Help" was clicked');
    });
    // also you can catch "onHelp" event from button
    // in this case you should attach "onClick" event to button
    dhxWins.window(id).button("help").attachEvent("onClick", function() {
        alert('button "Help" was clicked');
    });
</script>
