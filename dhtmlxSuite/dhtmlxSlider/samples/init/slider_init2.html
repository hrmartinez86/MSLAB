<div id="sliderBox"></div>
<script>
 
    //slider can be initialized by object notation
    var slider1 = new dhtmlxSlider("sliderBox", {
        skin: "ball",
        min:100,
        max:300,
        step:50,
        size:600,
        vertical:false
    });
    slider1.init();
</script>
