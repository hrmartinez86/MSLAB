<div id="sliderBox"></div>
<script>
 
    //slider can be initialized in specified container
    var slider1 = new dhtmlxSlider("sliderBox", 100);
    slider1.setImagePath("../../codebase/imgs/");
    slider1.init();    
    // first parametr - parent node, in case of direct initialization must be set to null
    // second parametr - width of slider
</script>
 
<script>
 
    //slider can be initialized directly in page , on place of javascript code
    var slider2 = new dhtmlxSlider(null, 200);
    slider2.init();
    // first parametr - parent node, in case of direct initialization must be set to null
</script>
