        ...
        mygrid.attachEvent("onRowSelect",doOnRowSelected);
        mygrid.attachEvent("onEditCell",doOnCellEdit);

        mygrid.attachEvent("onCheckbox",doOnCheck);
        mygrid.attachEvent("onBeforeRowDeleted",doBeforeRowDeleted);
        
        mygrid.init();
