Sample: dhtmlxWindows Global Events dhtmlxWindows main page
Resizing, Moving, Focusing 
 
<script>
 
    var dhxWins = new dhtmlXWindows();
    ...
    // catch focus
    dhxWins.attachEvent("onFocus", function(win) {
        alert("window was focused");
    });
    // catch moving
    dhxWins.attachEvent("onMoveFinish", function(win) {
        alert("window was moved");
    });
    // catch resizing
    dhxWins.attachEvent("onResizeFinish", function(win) {
        alert("window was resized");
    });
</script>
