Sample: dhtmlxColorPicker Using API dhtmlxColorPicker main page 
 
<script>
 
    //initialization new instance of dhtmlXColorPicker
    var _cpx = new dhtmlXColorPicker();
</script>
 
<script>
 
    //position of colorPicker
    var z=new dhtmlXColorPicker("null, true");
    z.init();
    z.setPosition(200,200);
</script>
 
<script>
 
    //selected color
    var z=new dhtmlXColorPicker("null, true");
    z.init();
    z.setColor("#123456");
    z.setColorRGB([56,34,12]);
</script>
 
<script>
 
    //hide - show
    var z=new dhtmlXColorPicker("null, true");
    z.init();
    z.hide();
    z.show();
    z.close();
</script>
 
<script>
 
    function init4() {
    //get selected color

    z.init();

    }
</script>
