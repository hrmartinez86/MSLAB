Sample: dhtmlxSlider Linking with input dhtmlxSlider main page
DHTML JavaScript Slider samples


 
<input type="text" id="input1">
<script>
 
    //slider can be linked to HTML input
    var slider1 = new dhtmlxSlider(null, 500);
    slider1.linkTo('input1');
    slider1.init();
</script>


 
<textarea id="textarea1"></textarea>
<script>
 
    //slider can be linked to HTML textarea
    var slider2=new dhtmlxSlider(null, 500);
    slider2.linkTo('textarea1');
    slider2.init();
</script>

</script>
