<div id="a1"></div>
<script>
 
    //slider onChange handler
    function my_func(pos,slider){
        document.getElementById("a1").innerHTML="Slider value is "+pos;
    }
 
    var slider=new dhtmlxSlider(null, 500);
    slider.attachEvent("onChange",my_func);
    slider.init();
</script>
